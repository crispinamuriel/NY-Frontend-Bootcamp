git -v = check git version 
git init = initalize git repository
git config --global user.name = set username
git config --global user.email = set email
git add <filename> = add to staging area
git add . = add all changed files to staging area
git commit -m 'message here' = commit your git changes to repository
git status = check status of files
git log = shows all commits 
git log --oneline = shows a small version of commits 
git rm --cached index.txt = remove untracked file from staging area
git log --all = shows all commits 
git checkout master = checkout master branch
git branch -a = ??
git branch newFeature = checkout to a new feature branch!
git branch -a = show the branches
git merge newFeature = merge new feature branch
